50#define vtkWarningToMessageCollectionWithObjectMacro(self, messageCollection, devMsgPrefix, userMsg) \
51 { \
52 vtkOStreamWrapper::EndlType endl; \
53 vtkOStreamWrapper::UseEndl(endl); \
54 if (messageCollection) \
