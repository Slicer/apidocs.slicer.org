88 inline void normcrossprod(double *v1, double *v2, double *norm)
89 // calculate normalized crossproduct
90 {

92 
93  norm[0] = v1[1] * v2[2] - v1[2] * v2[1];
94  norm[1] = v1[2] * v2[0] - v1[0] * v2[2];
