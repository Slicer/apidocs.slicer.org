77  for( int i = 0; i < 3; i++ )

79  mat[i][i] = ( eigenValues[i] < 0 ? -eigenValues[i] : eigenValues[i] );
80  }

82  matcorrect = eigenVectors * mat * eigenVectors.GetInverse();
83  tensorDouble.SetTensorFromMatrix( matcorrect );
